# 重做1.4.1的课后习题

```c++
#include <iostream>
#include <cstdlib>

using std::cin;
using std::cout;
using std::endl;

int main()
{
	int i, j;
	cout << "Please enter two integers:" << endl;
	cin >> i >> j;
	while (i <= j) {
		cout << i << endl;
		++i;
	}
	system("pause");
	return 0;
}
```
